```
cat_dreams/
├── main.py                  # основной файл
├── config.py                # константы: физика, цвета, настройки экрана, пути
│
├── entities/                # Игровые сущности
│   ├── cat.py               # Игрок: физика, коллизии, управление, машина состояний анимаций
│   ├── enemy.py             # Враги: патрулирование, атака, реакция на игрока
│   ├── camera.py            # Система приближения комнат
│   └── animation.py         # ️ компонент анимации спрайтов
│
├── managers/                # ️ Глобальные менеджеры
│   └── state_manager.py     # Управление сценами (Главное меню -> Игра -> Пауза) 
│
├── algorithms/              # Алгоритмы игры
│   ├── bsp_generator.py     # BSP-алгоритм: рекурсивное деление уровня на комнаты
│   ├── bfs_validator.py     # Валидация: проверка связности графа комнат (поиск пути)
│   └── pathfinder.py        # ️ A* алгоритм: поиск пути для врагов
│
└── assets/                  #  Ресурсы
    └── sprites/             # Спрайт-листы персонажей, врагов и тайлов
```
